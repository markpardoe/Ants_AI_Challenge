def move_distance(point1, point2)

        dR = (point1[0] - point2[0]).abs

        dC = (point1[1]-point2[1]).abs

        # Deal with wraparound map

        dR = @rows - dR if (dR*2 > @rows)

        dC = @cols - dC if (dC*2 > @cols)



        return dR + dC

end
